  });

  let html=`<div class="progress-wrap"><div class="progress-bar" style="width:${pct}%"></div></div>
  <div class="progress-meta"><span>${answered} of ${total} answered</span><span>${pct}% complete</span></div>`;

  Object.entries(sections).forEach(([name,info])=>{
    if(!info.concepts.length)return;
    html+=`<div class="section-header">
      <span class="section-badge ${info.badge}">${name==="Science (Optional)"?"OPTIONAL":name.toUpperCase()}</span>
      <h2>${name==="Reading"?"Reading &amp; Literature":name==="Grammar & Language"?"Grammar &amp; Language":name==="Math"?"Mathematics":"Science Reasoning — Bonus"}</h2>
    </div>`;
    info.concepts.forEach((c,ci)=>{
      html+=`<div class="concept-block">
        <div class="concept-title">${c.concept}</div>
        <div class="question-grid">
          <div class="question-col">
            <div class="col-header"><span class="col-tag tag-sat">SAT</span> Digital SAT</div>
            ${renderDiff(c.sat.easy,"easy")}${renderDiff(c.sat.medium,"medium")}${renderDiff(c.sat.hard,"hard")}
          </div>
          <div class="question-col">
            <div class="col-header"><span class="col-tag tag-act">ACT</span> Revised ACT</div>
            ${renderDiff(c.act.easy,"easy")}${renderDiff(c.act.medium,"medium")}${renderDiff(c.act.hard,"hard")}
          </div>
        </div>
        ${(name==="Grammar & Language"&&ci===info.concepts.length-1)?renderTimingRow("verbal"):""}
        ${(name==="Math"&&ci===info.concepts.length-1)?renderTimingRow("math"):""}
      </div>`;
    });
  });

  html+=`<div class="submit-zone">
    <button class="submit-btn" data-a="submit" ${answered===0?"disabled":""}>Submit &amp; See Results</button>
    <p style="margin-top:.6rem;font-size:12px;color:#aaa;">Unanswered questions are marked as skipped</p>
  </div>`;
  return html;
}

function renderTimingRow(sec){
  const opts=TIMING_OPTS[sec];
  const label=sec==="verbal"?"Reading &amp; Grammar section time:":"Math section time:";
  return`<div class="timing-row">
    <span class="timing-label">${label}</span>
    <div class="timing-options">
      ${opts.map((o,i)=>`<button class="timing-btn${S.timing[sec]===i?" selected":""}" data-a="timing" data-s="${sec}" data-i="${i}">${o.l}</button>`).join("")}
    </div>
  </div>`;
}

function renderDiff(q,diff){
  if(!q)return`<div class="diff-row"><div class="diff-label"><span class="diff-dot dot-${diff}"></span>${diff[0].toUpperCase()+diff.slice(1)} — not available</div></div>`;
  const sel=S.answers[q.id];
  const letters=q.options.length===5?["A","B","C","D","E"]:["A","B","C","D"];
  return`<div class="diff-row">
    <div class="diff-label"><span class="diff-dot dot-${diff}"></span>${diff[0].toUpperCase()+diff.slice(1)} &nbsp;·&nbsp; ${q.points}pt</div>
    ${q.passage?`<div class="q-passage">${q.passage.replace(/\n/g,"<br>")}</div>`:""}
    <div class="q-text">${q.question}</div>
    <div class="options">
      ${q.options.map((opt,i)=>{const l=letters[i];const s=sel===l;return`<button class="option-btn${s?" selected":""}" data-a="ans" data-id="${q.id}" data-v="${l}"><span class="opt-letter">${l}.</span>${opt}</button>`;}).join("")}
    </div>
  </div>`;
}

function renderResults(){
  const concepts=allConcepts();
  let vE=0,vM=0,mE=0,mM=0;
  const cResults=[];
  const tb={verbal:S.timing.verbal!==null?TIMING_OPTS.verbal[S.timing.verbal].v:0,math:S.timing.math!==null?TIMING_OPTS.math[S.timing.math].v:0};

  concepts.forEach(c=>{
    const isMath=c.section==="Math";
    const dots=[];let ce=0,cm=0;
    ["easy","medium","hard"].forEach(d=>{
      ["sat","act"].forEach(t=>{
        const q=c[t][d];if(!q)return;
        cm+=q.points;
        if(isMath)mM+=q.points; else vM+=q.points;
        const ans=S.answers[q.id];
        if(ans===undefined){dots.push("s");return;}
        if(ans===q.answer){ce+=q.points;if(isMath)mE+=q.points;else vE+=q.points;dots.push("c");}
        else dots.push("i");
      });
    });
    const pct=cm>0?Math.round(ce/cm*100):0;
    cResults.push({name:c.concept,section:c.section,pct,dots,ce,cm});
  });

  const vAdj=Math.max(0,vE+tb.verbal);
  const mAdj=Math.max(0,mE+tb.math);
  const vRange=getRange(vAdj,SAT_V_RANGE);
  const mRange=getRange(mAdj,SAT_M_RANGE);

  const strong=cResults.filter(r=>r.pct>=70).sort((a,b)=>b.pct-a.pct);
  const growth=cResults.filter(r=>r.pct<70).sort((a,b)=>a.pct-b.pct);

  function pC(p){return p>=70?"pct-s":p>=40?"pct-o":"pct-w";}
  function cRow(r){
    const dots=r.dots.map(d=>`<span class="cr-dot ${d}"></span>`).join("");
    return`<div class="concept-result-row">
      <span class="cr-section">${r.section}</span>
      <span class="cr-name">${r.name}</span>
      <div class="cr-dots">${dots}</div>
      <span class="cr-pct ${pC(r.pct)}">${r.pct}%</span>
    </div>`;
  }

  let easy={c:0,t:0},med={c:0,t:0},hard={c:0,t:0};
  concepts.forEach(c=>{
    ["sat","act"].forEach(t=>{
      ["easy","medium","hard"].forEach(d=>{
        const q=c[t][d];if(!q)return;
        const b=d==="easy"?easy:d==="medium"?med:hard;
        b.t++;if(S.answers[q.id]===q.answer)b.c++;
      });
    });
  });

  function bar(label,d){
    const p=d.t?Math.round(d.c/d.t*100):0;
    const col=p>=70?"#4a9e3f":p>=40?"#c47a15":"#b03030";
    return`<div class="diff-bar-row">
      <div class="diff-bar-meta"><span>${label}</span><span style="color:${col};font-weight:500">${p}% (${d.c}/${d.t})</span></div>
      <div class="diff-bar-track"><div class="diff-bar-fill" style="width:${p}%;background:${col}"></div></div>
    </div>`;
  }

  return`<div class="results-page">
    <div class="results-header">
      <h1>Diagnostic Results</h1>
      <p>Here's how you performed across all concepts and where to focus next.</p>
    </div>
    <div class="score-cards">
      <div class="score-card highlight">
        <div class="sc-label">SAT Verbal (adjusted)</div>
        <div class="sc-value">${vAdj.toFixed(1)}</div>
        <div class="sc-sub">pts · est. ${vRange}</div>
      </div>
      <div class="score-card highlight">
        <div class="sc-label">SAT Math (adjusted)</div>
        <div class="sc-value">${mAdj.toFixed(1)}</div>
        <div class="sc-sub">pts · est. ${mRange}</div>
      </div>
      <div class="score-card">
        <div class="sc-label">Verbal timing bonus</div>
        <div class="sc-value" style="font-size:20px">${tb.verbal>=0?"+":""}${tb.verbal}</div>
        <div class="sc-sub">${S.timing.verbal!==null?TIMING_OPTS.verbal[S.timing.verbal].l:"not recorded"}</div>
      </div>
      <div class="score-card">
        <div class="sc-label">Math timing bonus</div>
        <div class="sc-value" style="font-size:20px">${tb.math>=0?"+":""}${tb.math}</div>
        <div class="sc-sub">${S.timing.math!==null?TIMING_OPTS.math[S.timing.math].l:"not recorded"}</div>
      </div>
    </div>
    <div class="results-section">
      <h3>Areas of strength <span style="font-size:13px;font-weight:400;color:#999;">(≥70% correct)</span></h3>
      ${strong.length?strong.map(cRow).join(""):`<p style="font-size:13px;color:#888;padding:.5rem 0;">No concepts at mastery level yet — that's what we're here to work on.</p>`}
    </div>
    <div class="results-section">
      <h3>Growth areas <span style="font-size:13px;font-weight:400;color:#999;">(&lt;70% correct)</span></h3>
      ${growth.length?growth.map(cRow).join(""):`<p style="font-size:13px;color:#888;padding:.5rem 0;">Strong across all concepts — excellent diagnostic performance.</p>`}
    </div>
    <div class="results-section">
      <h3>Difficulty breakdown</h3>
      <div class="diff-breakdown">${bar("Easy questions",easy)}${bar("Medium questions",med)}${bar("Hard questions",hard)}</div>
    </div>
    <div class="dot-legend">
      <strong style="font-size:13px;">Dot key</strong>
      <div class="dot-legend-row">
        <span style="display:flex;align-items:center;gap:5px;"><span style="width:10px;height:10px;border-radius:50%;background:#4a9e3f;display:inline-block;"></span>Correct</span>
        <span style="display:flex;align-items:center;gap:5px;"><span style="width:10px;height:10px;border-radius:50%;background:#e04040;display:inline-block;"></span>Incorrect</span>
        <span style="display:flex;align-items:center;gap:5px;"><span style="width:10px;height:10px;border-radius:50%;background:#ccc;display:inline-block;"></span>Skipped</span>
        <span style="color:#aaa;">Left to right: SAT Easy · ACT Easy · SAT Medium · ACT Medium · SAT Hard · ACT Hard</span>
      </div>
    </div>
    <div class="retry-zone">
      <button class="btn-outline" data-a="retry">Retake Diagnostic</button>
      <button class="btn-outline" data-a="review">Review My Answers</button>
    </div>
  </div>`;
}

function bind(){
  document.querySelectorAll("[data-a]").forEach(el=>{
    el.onclick=e=>{
      e.stopPropagation();
      const a=el.dataset.a;
      if(a==="start"){S.page="test";render();scrollTo(0,0);}
      else if(a==="sci"){S.showSci=!S.showSci;render();}
      else if(a==="ans"){
        const prev=S.answers[el.dataset.id];
        if(prev===el.dataset.v) delete S.answers[el.dataset.id];
        else S.answers[el.dataset.id]=el.dataset.v;
        render();
        setTimeout(()=>{const b=document.querySelector(`[data-id="${el.dataset.id}"]`);if(b)b.scrollIntoView({block:"nearest",behavior:"smooth"});},50);
      }
      else if(a==="timing"){S.timing[el.dataset.s]=parseInt(el.dataset.i);render();}
      else if(a==="submit"){S.page="results";render();scrollTo(0,0);}
      else if(a==="retry"){S={page:"test",answers:{},timing:{verbal:null,math:null},showSci:S.showSci};render();scrollTo(0,0);}
      else if(a==="review"){S.page="test";render();scrollTo(0,0);}
    };
  });
}

render();
